)

    test_ArrayView2D.cpp
    test_ArrayView2DAlgorithms.cpp
    test_Index2DMap.cpp
    test_IntervalMap.cpp
    test_MutableArrayView2D.cpp
    test_RingBuffer.cpp
)

    PRIVATE
        Catch2::Catch2WithMain
        Chorasmia::Chorasmia
)
